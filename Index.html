<script type="text/babel">
    const { useState, useEffect, useRef } = React;

    const Menu = () => <svg className="icon w-6 h-6"><line x1="3" y1="12" x2="21" y2="12"/><line x1="3" y1="6" x2="21" y2="6"/><line x1="3" y1="18" x2="21" y2="18"/></svg>;
    const X = () => <svg className="icon w-6 h-6"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>;
    const Search = () => <svg className="icon w-5 h-5"><circle cx="11" cy="11" r="8"/><path d="m21 21-4.35-4.35"/></svg>;
    const RefreshCw = () => <svg className="icon w-5 h-5"><polyline points="23 4 23 10 17 10"/><polyline points="1 20 1 14 7 14"/><path d="M3.51 9a9 9 0 0 1 14.85-3.36L23 10M1 14l4.64 4.36A9 9 0 0 0 20.49 15"/></svg>;
    const Plus = () => <svg className="icon w-5 h-5"><line x1="12" y1="5" x2="12" y2="19"/><line x1="5" y1="12" x2="19" y2="12"/></svg>;
    const Play = () => <svg className="icon w-5 h-5"><polygon points="5 3 19 12 5 21 5 3"/></svg>;
    const TrendingUp = () => <svg className="icon w-5 h-5"><polyline points="23 6 13.5 15.5 8.5 10.5 1 18"/><polyline points="17 6 23 6 23 12"/></svg>;
    const TrendingDown = () => <svg className="icon w-5 h-5"><polyline points="23 18 13.5 8.5 8.5 13.5 1 6"/><polyline points="17 18 23 18 23 12"/></svg>;
    const Zap = () => <svg className="icon w-5 h-5"><polygon points="13 2 3 14 12 14 11 22 21 10 12 10 13 2"/></svg>;
    const Trash2 = () => <svg className="icon w-4 h-4"><polyline points="3 6 5 6 21 6"/><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"/><line x1="10" y1="11" x2="10" y2="17"/><line x1="14" y1="11" x2="14" y2="17"/></svg>;
    const CheckCircle = () => <svg className="icon w-5 h-5"><path d="M22 11.08V12a10 10 0 1 1-5.93-9.14"/><polyline points="22 4 12 14.01 9 11.01"/></svg>;
    const AlertCircle = () => <svg className="icon w-5 h-5"><circle cx="12" cy="12" r="10"/><line x1="12" y1="8" x2="12" y2="12"/><line x1="12" y1="16" x2="12.01" y2="16"/></svg>;
    const Database = () => <svg className="icon w-5 h-5"><ellipse cx="12" cy="5" rx="9" ry="3"/><path d="M21 12c0 1.66-4 3-9 3s-9-1.34-9-3"/><path d="M3 5v14c0 1.66 4 3 9 3s9-1.34 9-3V5"/></svg>;
    const DollarSign = () => <svg className="icon w-5 h-5"><line x1="12" y1="1" x2="12" y2="23"/><path d="M17 5H9.5a3.5 3.5 0 0 0 0 7h5a3.5 3.5 0 0 1 0 7H6"/></svg>;
    const BarChart3 = () => <svg className="icon w-7 h-7"><path d="M3 3v18h18"/><path d="M18 17V9"/><path d="M13 17V5"/><path d="M8 17v-3"/></svg>;

    const TradingPlatform = () => {
        const [activeTab, setActiveTab] = useState('home');
        const [mobileMenuOpen, setMobileMenuOpen] = useState(false);

        // API Configuration - Stored in localStorage
        const [apiConfig, setApiConfig] = useState(() => {
            const saved = localStorage.getItem('apiConfig');
            return saved ? JSON.parse(saved) : {
                provider: 'alphavantage',
                apiKey: '',
                enabled: false
            };
        });

        // Save API config to localStorage
        useEffect(() => {
            localStorage.setItem('apiConfig', JSON.stringify(apiConfig));
        }, [apiConfig]);

        // Portfolio State - Stored in localStorage
        const [portfolio, setPortfolio] = useState(() => {
            const saved = localStorage.getItem('portfolio');
            return saved ? JSON.parse(saved) : [
                { symbol: 'AAPL', name: 'Apple Inc', shares: 100, avgPrice: 150.00, currentPrice: 185.50, sector: 'Technology', beta: 1.2, volatility: 0.22 },
                { symbol: 'MSFT', name: 'Microsoft', shares: 50, avgPrice: 300.00, currentPrice: 402.50, sector: 'Technology', beta: 1.0, volatility: 0.20 },
                { symbol: 'NVDA', name: 'NVIDIA', shares: 75, avgPrice: 80.00, currentPrice: 495.30, sector: 'Technology', beta: 2.4, volatility: 0.49 },
            ];
        });

        // Save portfolio to localStorage
        useEffect(() => {
            localStorage.setItem('portfolio', JSON.stringify(portfolio));
        }, [portfolio]);

        const [newStock, setNewStock] = useState({ symbol: '', shares: '', avgPrice: '' });
        const [lookupLoading, setLookupLoading] = useState(false);
        const [editingStock, setEditingStock] = useState(null);

        // Screener state
        const [screenerCriteria, setScreenerCriteria] = useState({
            minPrice: 0,
            maxPrice: 10000,
            minMarketCap: 0,
            maxMarketCap: 5000,
            technicalPattern: 'all',
        });
        const [screenerResults, setScreenerResults] = useState([]);
        const [screenerLoading, setScreenerLoading] = useState(false);
        const [searchSymbols, setSearchSymbols] = useState('');

        // Backtesting state
        const [backtestConfig, setBacktestConfig] = useState({
            strategy: 'rsi_oversold',
            initialCapital: 10000,
            positionSize: 0.1,
            stopLoss: 0.05,
            takeProfit: 0.10,
            rsiBuyThreshold: 30,
            rsiSellThreshold: 70,
        });
        const [backtestResults, setBacktestResults] = useState(null);
        const [backtestRunning, setBacktestRunning] = useState(false);

        // Trading signals
        const [tradingSignals, setTradingSignals] = useState([]);

        // ========== UNIVERSAL API FUNCTIONS ==========

        // Fetch quote for ANY stock
        const fetchUniversalQuote = async (symbol) => {
            if (!apiConfig.enabled || !apiConfig.apiKey) {
                console.log('API not configured');
                return null;
            }

            try {
                let quote = null;

                if (apiConfig.provider === 'alphavantage') {
                    const response = await fetch(
                        `https://www.alphavantage.co/query?function=GLOBAL_QUOTE&symbol=${symbol}&apikey=${apiConfig.apiKey}`
                    );
                    const data = await response.json();

                    if (data['Global Quote'] && Object.keys(data['Global Quote']).length > 0) {
                        const q = data['Global Quote'];
                        quote = {
                            symbol,
                            name: symbol,
                            price: parseFloat(q['05. price']),
                            change: parseFloat(q['09. change']),
                            changePercent: parseFloat(q['10. change percent'].replace('%', '')),
                            volume: parseInt(q['06. volume']),
                            high: parseFloat(q['03. high']),
                            low: parseFloat(q['04. low']),
                        };
                    } else if (data['Note']) {
                        alert('API rate limit reached. Please wait 1 minute or use Finnhub.');
                        return null;
                    }
                } else if (apiConfig.provider === 'finnhub') {
                    const response = await fetch(
                        `https://finnhub.io/api/v1/quote?symbol=${symbol}&token=${apiConfig.apiKey}`
                    );
                    const data = await response.json();

                    if (data.c > 0) {
                        quote = {
                            symbol,
                            name: symbol,
                            price: data.c,
                            change: data.d,
                            changePercent: data.dp,
                            high: data.h,
                            low: data.l,
                        };
                    }
                }

                return quote;
            } catch (error) {
                console.error('Error fetching quote:', error);
                return null;
            }
        };

        // Search for ANY stock symbol
        const searchStocks = async (query) => {
            if (!query || query.length < 1) return [];
            if (!apiConfig.enabled || !apiConfig.apiKey) {
                alert('Please configure API in Settings to search stocks');
                return [];
            }

            setScreenerLoading(true);
            const symbols = query.split(',').map(s => s.trim().toUpperCase()).filter(s => s.length > 0);
            const results = [];

            for (const symbol of symbols) {
                const quote = await fetchUniversalQuote(symbol);
                if (quote) {
                    results.push({
                        ...quote,
                        marketCap: 0,
                        pe: 0,
                        dividendYield: 0,
                        beta: 1.0,
                        rsi: 50,
                        sector: 'Unknown'
                    });
                }
            }

            setScreenerResults(results);
            setScreenerLoading(false);
            return results;
        };

        // Auto-lookup when adding stock
        const handleSymbolLookup = async (symbol) => {
            if (symbol.length < 1) return;

            setLookupLoading(true);
            const quote = await fetchUniversalQuote(symbol.toUpperCase());

            if (quote) {
                setNewStock(prev => ({
                    ...prev,
                    symbol: symbol.toUpperCase(),
                    avgPrice: quote.price.toFixed(2)
                }));
            }
            setLookupLoading(false);
        };

        // Update portfolio prices
        const updateAllPrices = async () => {
            if (!apiConfig.enabled || !apiConfig.apiKey) {
                alert('Please configure API in Settings to update prices');
                return;
            }

            for (const stock of portfolio) {
                const quote = await fetchUniversalQuote(stock.symbol);
                if (quote) {
                    setPortfolio(prev => prev.map(s =>
                        s.symbol === stock.symbol ? { ...s, currentPrice: quote.price } : s
                    ));
                }
                // Add small delay to avoid rate limiting
                await new Promise(resolve => setTimeout(resolve, 1000));
            }
        };

        // Add stock
        const addStock = async () => {
            if (!newStock.symbol || !newStock.shares || !newStock.avgPrice) {
                alert('Please fill all fields');
                return;
            }

            const stock = {
                symbol: newStock.symbol.toUpperCase(),
                name: newStock.symbol.toUpperCase(),
                shares: parseFloat(newStock.shares),
                avgPrice: parseFloat(newStock.avgPrice),
                currentPrice: parseFloat(newStock.avgPrice),
                sector: 'Technology',
                beta: 1.0,
                volatility: 0.25
            };

            setPortfolio([...portfolio, stock]);
            setNewStock({ symbol: '', shares: '', avgPrice: '' });
            setEditingStock(null);
        };

        const removeStock = (index) => {
            if (window.confirm('Remove this stock?')) {
                setPortfolio(portfolio.filter((_, i) => i !== index));
            }
        };

        // Backtesting
        const runBacktest = () => {
            setBacktestRunning(true);
            setTimeout(() => {
                const results = performBacktest();
                setBacktestResults(results);
                setBacktestRunning(false);
            }, 2000);
        };

        const performBacktest = () => {
            let capital = backtestConfig.initialCapital;
            let shares = 0;
            let trades = [];
            let equity = [capital];
            const days = 250;

            let price = 180;
            const prices = [];
            for (let i = 0; i < days + 200; i++) {
                price *= (1 + (Math.random() - 0.48) * 0.02);
                prices.push(price);
            }

            const rsiValues = [];
            for (let d = 0; d < days; d++) {
                const day = d + 200;
                let gains = 0, losses = 0;
                for (let i = day - 14; i < day; i++) {
                    const change = prices[i] - prices[i - 1];
                    if (change > 0) gains += change;
                    else losses -= change;
                }
                const rs = (gains / 14) / (losses / 14);
                rsiValues.push(100 - (100 / (1 + rs)));
            }

            for (let d = 0; d < days; d++) {
                const p = prices[d + 200];
                const rsi = rsiValues[d];

                if (rsi < backtestConfig.rsiBuyThreshold && capital > 0 && shares === 0) {
                    const invest = capital * backtestConfig.positionSize;
                    const buyShares = Math.floor(invest / p);
                    if (buyShares > 0) {
                        shares = buyShares;
                        capital -= buyShares * p;
                        trades.push({ type: 'BUY', price: p.toFixed(2), shares: buyShares });
                    }
                } else if (shares > 0) {
                    const entryPrice = parseFloat(trades[trades.length - 1].price);
                    const stopPrice = entryPrice * (1 - backtestConfig.stopLoss);
                    const targetPrice = entryPrice * (1 + backtestConfig.takeProfit);

                    if (p <= stopPrice || p >= targetPrice || rsi > backtestConfig.rsiSellThreshold) {
                        capital += shares * p;
                        const profit = (p - entryPrice) * shares;
                        trades.push({ type: 'SELL', price: p.toFixed(2), shares, profit: profit.toFixed(2) });
                        shares = 0;
                    }
                }

                equity.push(capital + shares * p);
            }

            if (shares > 0) {
                capital += shares * prices[prices.length - 1];
            }

            const finalEquity = capital;
            const totalReturn = ((finalEquity - backtestConfig.initialCapital) / backtestConfig.initialCapital * 100);
            const sellTrades = trades.filter(t => t.type === 'SELL');
            const winTrades = sellTrades.filter(t => parseFloat(t.profit) > 0);

            return {
                finalEquity: finalEquity.toFixed(2),
                totalReturn: totalReturn.toFixed(2),
                totalTrades: sellTrades.length,
                winningTrades: winTrades.length,
                winRate: sellTrades.length > 0 ? ((winTrades.length / sellTrades.length) * 100).toFixed(1) : 0,
                trades: trades.slice(-20)
            };
        };

        // Generate signals
        const generateSignals = () => {
            const signals = portfolio.map(stock => {
                const rsi = 50 + (Math.random() - 0.5) * 40;
                let signal = 'HOLD';
                let confidence = 50;
                let reason = 'No clear signal';

                if (rsi < 30) {
                    signal = 'BUY';
                    confidence = 75;
                    reason = `RSI oversold (${rsi.toFixed(1)})`;
                } else if (rsi > 70) {
                    signal = 'SELL';
                    confidence = 75;
                    reason = `RSI overbought (${rsi.toFixed(1)})`;
                }

                return {
                    ...stock,
                    signal,
                    confidence: confidence.toFixed(0),
                    reason,
                    target: signal === 'BUY' ? stock.currentPrice * 1.05 : stock.currentPrice * 0.95,
                    stopLoss: signal === 'BUY' ? stock.currentPrice * 0.97 : stock.currentPrice * 1.03,
                };
            });
            setTradingSignals(signals);
        };

        useEffect(() => {
            generateSignals();
        }, [portfolio]);

        // Calculate metrics
        const metrics = portfolio.reduce((acc, stock) => {
            const value = stock.shares * stock.currentPrice;
            const cost = stock.shares * stock.avgPrice;
            return {
                totalValue: acc.totalValue + value,
                totalCost: acc.totalCost + cost,
                totalGainLoss: acc.totalGainLoss + (value - cost)
            };
        }, { totalValue: 0, totalCost: 0, totalGainLoss: 0 });

        const totalReturn = metrics.totalCost > 0 ? ((metrics.totalGainLoss / metrics.totalCost) * 100).toFixed(2) : '0.00';

        // Mobile nav
        const MobileNav = () => (
            <div className={`fixed inset-0 bg-slate-950 z-50 transform transition-transform duration-300 ${mobileMenuOpen ? 'translate-x-0' : 'translate-x-full'}`}>
                <div className="p-6">
                    <div className="flex justify-between items-center mb-8">
                        <h2 className="text-2xl font-bold text-white">Menu</h2>
                        <button onClick={() => setMobileMenuOpen(false)} className="text-white"><X /></button>
                    </div>
                    <div className="space-y-3">
                        {[
                            { id: 'home', label: 'Home', icon: Home },
                            { id: 'portfolio', label: 'Portfolio', icon: PieChart },
                            { id: 'screener', label: 'Screener', icon: Filter },
                            { id: 'backtest', label: 'Backtest', icon: Activity },
                            { id: 'signals', label: 'Signals', icon: Target },
                            { id: 'settings', label: 'Settings', icon: Settings },
                        ].map(item => (
                            <button key={item.id} onClick={() => { setActiveTab(item.id); setMobileMenuOpen(false); }}
                                className={`w-full flex items-center gap-3 p-4 rounded-lg ${activeTab === item.id ? 'bg-blue-500 text-white' : 'bg-white/5 text-gray-300'}`}>
                                <item.icon />
                                <span className="font-semibold">{item.label}</span>
                            </button>
                        ))}
                    </div>
                </div>
            </div>
        );

        return (
            <div className="min-h-screen bg-gradient-to-br from-slate-950 via-blue-950 to-indigo-950">

                {/* Header */}
                <div className="sticky top-0 z-40 bg-white/10 backdrop-blur-lg border-b border-white/20">
                    <div className="flex items-center justify-between p-4">
                        <div className="flex items-center gap-3">
                            <BarChart3 />
                            <div>
                                <h1 className="text-lg font-bold text-white">Trading Platform Pro</h1>
                                <p className="text-xs text-blue-200">Universal Stock Access</p>
                            </div>
                        </div>
                        <button onClick={() => setMobileMenuOpen(true)} className="text-white"><Menu /></button>
                    </div>
                </div>

                <MobileNav />

                <div className="p-4 pb-24 max-w-7xl mx-auto">

                    {/* HOME */}
                    {activeTab === 'home' && (
                        <div className="space-y-4">
                            <div className="bg-gradient-to-br from-blue-500/20 to-purple-500/20 rounded-2xl p-6 border border-blue-500/30">
                                <div className="text-blue-200 text-sm mb-2 flex items-center gap-2"><DollarSign />Portfolio Value</div>
                                <div className="text-white font-bold text-4xl mb-2">${metrics.totalValue.toFixed(2)}</div>
                                <div className="flex items-center gap-3">
                                    <div className={`text-xl font-semibold flex items-center gap-1 ${parseFloat(totalReturn) >= 0 ? 'text-green-400' : 'text-red-400'}`}>
                                        {parseFloat(totalReturn) >= 0 ? <TrendingUp /> : <TrendingDown />}
                                        {totalReturn}%
                                    </div>
                                    <div className="text-sm text-gray-300">${metrics.totalGainLoss.toFixed(2)}</div>
                                </div>
                                <button onClick={updateAllPrices} className="mt-4 flex items-center gap-1 text-xs bg-white/10 px-3 py-1 rounded-lg text-white">
                                    <RefreshCw />Update Prices
                                </button>
                            </div>

                            <div className="grid grid-cols-2 gap-3">
                                <div className="bg-white/5 rounded-xl p-4 border border-white/10">
                                    <div className="text-gray-400 text-xs mb-1">Holdings</div>
                                    <div className="text-white font-bold text-2xl">{portfolio.length}</div>
                                </div>
                                <div className="bg-white/5 rounded-xl p-4 border border-white/10">
                                    <div className="text-gray-400 text-xs mb-1">API Status</div>
                                    <div className={`font-bold text-xl ${apiConfig.enabled ? 'text-green-400' : 'text-yellow-400'}`}>
                                        {apiConfig.enabled ? 'LIVE' : 'OFFLINE'}
                                    </div>
                                </div>
                            </div>

                            <div className="bg-white/5 rounded-xl p-4 border border-white/10">
                                <h3 className="text-white font-semibold mb-3">Holdings</h3>
                                {portfolio.map((stock, i) => {
                                    const value = stock.shares * stock.currentPrice;
                                    const gain = ((stock.currentPrice - stock.avgPrice) / stock.avgPrice * 100).toFixed(2);
                                    return (
                                        <div key={i} className="flex justify-between py-3 border-b border-white/10 last:border-0">
                                            <div>
                                                <div className="text-white font-semibold">{stock.symbol}</div>
                                                <div className="text-gray-400 text-sm">${stock.currentPrice.toFixed(2)}</div>
                                            </div>
                                            <div className="text-right">
                                                <div className="text-white font-semibold">${value.toFixed(0)}</div>
                                                <div className={`text-sm ${parseFloat(gain) >= 0 ? 'text-green-400' : 'text-red-400'}`}>
                                                    {gain}%
                                                </div>
                                            </div>
                                        </div>
                                    );
                                })}
                            </div>
                        </div>
                    )}

                    {/* PORTFOLIO */}
                    {activeTab === 'portfolio' && (
                        <div className="space-y-4">
                            <div className="flex justify-between items-center">
                                <h2 className="text-2xl font-bold text-white">Portfolio</h2>
                                <button onClick={() => setEditingStock({ type: 'add' })} className="bg-green-500 text-white px-4 py-2 rounded-lg flex items-center gap-2">
                                    <Plus />Add
                                </button>
                            </div>

                            {editingStock?.type === 'add' && (
                                <div className="fixed inset-0 bg-black/50 z-50 flex items-center justify-center p-4">
                                    <div className="bg-slate-900 rounded-2xl p-6 max-w-md w-full border border-white/20">
                                        <h3 className="text-xl font-bold text-white mb-4">Add Any Stock</h3>
                                        <div className="space-y-3">
                                            <div>
                                                <input
                                                    type="text"
                                                    placeholder="Symbol (ANY stock: AAPL, TSLA, etc.)"
                                                    value={newStock.symbol}
                                                    onChange={(e) => setNewStock({...newStock, symbol: e.target.value})}
                                                    onBlur={(e) => handleSymbolLookup(e.target.value)}
                                                    className="w-full bg-white/10 text-white px-4 py-3 rounded-lg border border-white/20 uppercase"
                                                />
                                                {lookupLoading && <div className="text-blue-400 text-xs mt-1">Looking up price...</div>}
                                                {apiConfig.enabled && <div className="text-green-400 text-xs mt-1">âœ“ Live price lookup enabled</div>}
                                            </div>
                                            <input type="number" placeholder="Shares" value={newStock.shares}
                                                onChange={(e) => setNewStock({...newStock, shares: e.target.value})}
                                                className="w-full bg-white/10 text-white px-4 py-3 rounded-lg border border-white/20" />
                                            <input type="number" step="0.01" placeholder="Price" value={newStock.avgPrice}
                                                onChange={(e) => setNewStock({...newStock, avgPrice: e.target.value})}
                                                className="w-full bg-white/10 text-white px-4 py-3 rounded-lg border border-white/20" />
                                        </div>
                                        <div className="flex gap-3 mt-6">
                                            <button onClick={addStock} className="flex-1 bg-green-500 text-white py-3 rounded-lg font-semibold">Add</button>
                                            <button onClick={() => setEditingStock(null)} className="flex-1 bg-gray-600 text-white py-3 rounded-lg font-semibold">Cancel</button>
                                        </div>
                                    </div>
                                </div>
                            )}

                            {portfolio.map((stock, i) => (
                                <div key={i} className="bg-white/5 rounded-xl p-4 border border-white/10">
                                    <div className="flex justify-between mb-3">
                                        <div>
                                            <div className="text-xl font-bold text-white">{stock.symbol}</div>
                                            <div className="text-gray-400 text-sm">{stock.name}</div>
                                        </div>
                                        <button onClick={() => removeStock(i)} className="text-red-400"><Trash2 /></button>
                                    </div>
                                    <div className="grid grid-cols-3 gap-2 text-sm">
                                        <div><div className="text-gray-400 text-xs">Shares</div><div className="text-white font-semibold">{stock.shares}</div></div>
                                        <div><div className="text-gray-400 text-xs">Price</div><div className="text-white font-semibold">${stock.currentPrice.toFixed(2)}</div></div>
                                        <div><div className="text-gray-400 text-xs">Value</div><div className="text-white font-semibold">${(stock.shares * stock.currentPrice).toFixed(0)}</div></div>
                                    </div>
                                </div>
                            ))}
                        </div>
                    )}

                    {/* SCREENER */}
                    {activeTab === 'screener' && (
                        <div className="space-y-4">
                            <h2 className="text-2xl font-bold text-white">Universal Stock Search</h2>

                            <div className="bg-white/5 rounded-xl p-4 border border-white/10">
                                <label className="text-white font-semibold mb-2 block">Enter Stock Symbols (comma-separated)</label>
                                <input
                                    type="text"
                                    placeholder="AAPL, TSLA, GOOGL, MSFT, AMZN..."
                                    value={searchSymbols}
                                    onChange={(e) => setSearchSymbols(e.target.value)}
                                    className="w-full bg-white/10 text-white px-4 py-3 rounded-lg border border-white/20 mb-3"
                                />
                                <button
                                    onClick={() => searchStocks(searchSymbols)}
                                    disabled={screenerLoading}
                                    className="w-full bg-blue-500 text-white py-3 rounded-lg font-semibold flex items-center justify-center gap-2"
                                >
                                    {screenerLoading ? <><RefreshCw className="animate-spin" />Searching...</> : <><Search />Search Stocks</>}
                                </button>
                            </div>

                            {screenerResults.length > 0 && (
                                <div className="space-y-3">
                                    <h3 className="text-white font-semibold">{screenerResults.length} Results</h3>
                                    {screenerResults.map((stock, i) => (
                                        <div key={i} className="bg-white/5 rounded-xl p-4 border border-white/10">
                                            <div className="flex justify-between">
                                                <div>
                                                    <div className="text-lg font-bold text-white">{stock.symbol}</div>
                                                    <div className="text-gray-400 text-sm">{stock.name}</div>
                                                </div>
                                                <div className="text-right">
                                                    <div className="text-lg font-bold text-white">${stock.price.toFixed(2)}</div>
                                                    <div className={`text-sm ${stock.changePercent >= 0 ? 'text-green-400' : 'text-red-400'}`}>
                                                        {stock.changePercent >= 0 ? '+' : ''}{stock.changePercent.toFixed(2)}%
                                                    </div>
                                                </div>
                                            </div>
                                        </div>
                                    ))}
                                </div>
                            )}
                        </div>
                    )}

                    {/* BACKTEST */}
                    {activeTab === 'backtest' && (
                        <div className="space-y-4">
                            <h2 className="text-2xl font-bold text-white">Backtest Strategy</h2>

                            <div className="bg-white/5 rounded-xl p-4 border border-white/10 space-y-3">
                                <div>
                                    <label className="text-gray-300 text-sm mb-2 block">Initial Capital</label>
                                    <input type="number" value={backtestConfig.initialCapital}
                                        onChange={(e) => setBacktestConfig({...backtestConfig, initialCapital: parseFloat(e.target.value)})}
                                        className="w-full bg-white/10 text-white px-3 py-2 rounded-lg border border-white/20" />
                                </div>
                                <button onClick={runBacktest} disabled={backtestRunning}
                                    className="w-full bg-purple-500 text-white py-4 rounded-xl font-bold flex items-center justify-center gap-2">
                                    {backtestRunning ? <><RefreshCw className="animate-spin" />Running...</> : <><Play />Run Backtest</>}
                                </button>
                            </div>

                            {backtestResults && (
                                <div className="space-y-4">
                                    <div className="grid grid-cols-2 gap-3">
                                        <div className="bg-green-500/20 rounded-xl p-4 border border-green-500/30">
                                            <div className="text-green-300 text-xs mb-1">Total Return</div>
                                            <div className="text-2xl font-bold text-green-400">{backtestResults.totalReturn}%</div>
                                        </div>
                                        <div className="bg-blue-500/20 rounded-xl p-4 border border-blue-500/30">
                                            <div className="text-blue-300 text-xs mb-1">Final Equity</div>
                                            <div className="text-2xl font-bold text-white">${backtestResults.finalEquity}</div>
                                        </div>
                                        <div className="bg-white/5 rounded-xl p-4 border border-white/10">
                                            <div className="text-gray-400 text-xs mb-1">Win Rate</div>
                                            <div className="text-xl font-bold text-white">{backtestResults.winRate}%</div>
                                        </div>
                                        <div className="bg-white/5 rounded-xl p-4 border border-white/10">
                                            <div className="text-gray-400 text-xs mb-1">Trades</div>
                                            <div className="text-xl font-bold text-white">{backtestResults.totalTrades}</div>
                                        </div>
                                    </div>
                                </div>
                            )}
                        </div>
                    )}

                    {/* SIGNALS */}
                    {activeTab === 'signals' && (
                        <div className="space-y-4">
                            <div className="flex justify-between items-center">
                                <h2 className="text-2xl font-bold text-white">Trading Signals</h2>
                                <button onClick={generateSignals} className="bg-blue-500 text-white px-4 py-2 rounded-lg flex items-center gap-2">
                                    <RefreshCw />Refresh
                                </button>
                            </div>

                            {tradingSignals.map((signal, i) => (
                                <div key={i} className={`rounded-xl p-4 border-l-4 ${
                                    signal.signal === 'BUY' ? 'bg-green-500/10 border-green-500' :
                                    signal.signal === 'SELL' ? 'bg-red-500/10 border-red-500' :
                                    'bg-blue-500/10 border-blue-500'
                                }`}>
                                    <div className="flex justify-between mb-2">
                                        <div>
                                            <div className="text-xl font-bold text-white">{signal.symbol}</div>
                                            <div className="text-sm text-gray-300">{signal.reason}</div>
                                        </div>
                                        <div className={`px-4 py-2 rounded-lg font-bold ${
                                            signal.signal === 'BUY' ? 'bg-green-500' :
                                            signal.signal === 'SELL' ? 'bg-red-500' : 'bg-blue-500'
                                        } text-white`}>{signal.signal}</div>
                                    </div>
                                    <div className="grid grid-cols-3 gap-2 text-xs">
                                        <div><div className="text-gray-400">Target</div><div className="text-green-400 font-semibold">${signal.target.toFixed(2)}</div></div>
                                        <div><div className="text-gray-400">Stop</div><div className="text-red-400 font-semibold">${signal.stopLoss.toFixed(2)}</div></div>
                                        <div><div className="text-gray-400">Confidence</div><div className="text-white font-semibold">{signal.confidence}%</div></div>
                                    </div>
                                </div>
                            ))}
                        </div>
                    )}

                    {/* SETTINGS */}
                    {activeTab === 'settings' && (
                        <div className="space-y-4">
                            <h2 className="text-2xl font-bold text-white">API Settings</h2>

                            <div className="bg-blue-500/10 border border-blue-500/30 rounded-xl p-4">
                                <h3 className="text-blue-300 font-semibold mb-2 flex items-center gap-2"><Zap />Enable Universal Stock Access</h3>
                                <p className="text-blue-200 text-sm mb-3">Get FREE API key for unlimited stock access</p>

                                <div className="space-y-3">
                                    <div>
                                        <label className="text-white font-semibold mb-2 block text-sm">Provider</label>
                                        <select value={apiConfig.provider} onChange={(e) => setApiConfig({...apiConfig, provider: e.target.value})}
                                            className="w-full bg-white/10 text-white px-3 py-2 rounded-lg border border-white/20">
                                            <option value="alphavantage">Alpha Vantage (25 calls/day)</option>
                                            <option value="finnhub">Finnhub (60 calls/min)</option>
                                        </select>
                                    </div>
                                    <div>
                                        <label className="text-white font-semibold mb-2 block text-sm">API Key</label>
                                        <input type="text" value={apiConfig.apiKey}
                                            onChange={(e) => setApiConfig({...apiConfig, apiKey: e.target.value})}
                                            placeholder="Enter your API key"
                                            className="w-full bg-white/10 text-white px-3 py-2 rounded-lg border border-white/20" />
                                    </div>
                                    <label className="flex items-center gap-2">
                                        <input type="checkbox" checked={apiConfig.enabled}
                                            onChange={(e) => setApiConfig({...apiConfig, enabled: e.target.checked})}
                                            className="w-5 h-5" />
                                        <span className="text-white font-semibold">Enable Real-Time Data</span>
                                    </label>
                                    {apiConfig.enabled && apiConfig.apiKey && (
                                        <div className="bg-green-500/10 border border-green-500/30 rounded-lg p-3">
                                            <div className="flex items-center gap-2 text-green-400"><CheckCircle />API Connected!</div>
                                        </div>
                                    )}
                                </div>
                            </div>

                            <div className="bg-white/5 rounded-xl p-4 border border-white/10">
                                <h3 className="text-white font-semibold mb-2">Get FREE API Key:</h3>
                                <div className="space-y-2 text-sm text-gray-300">
                                    <div><strong className="text-white">Alpha Vantage:</strong><br/><https://www.alphavantage.co/support/#api-key></div>
                                    <div><strong className="text-white">Finnhub:</strong><br/><https://finnhub.io/register></div>
                                </div>
                            </div>
                        </div>
                    )}
                </div>

                {/* Bottom Nav */}
                <div className="fixed bottom-0 left-0 right-0 bg-white/10 backdrop-blur-lg border-t border-white/20">
                    <div className="flex justify-around py-2">
                        {[
                            { id: 'home', icon: Home, label: 'Home' },
                            { id: 'portfolio', icon: PieChart, label: 'Portfolio' },
                            { id: 'screener', icon: Filter, label: 'Search' },
                            { id: 'backtest', icon: Activity, label: 'Test' },
                            { id: 'signals', icon: Target, label: 'Signals' },
                        ].map(item => (
                            <button key={item.id} onClick={() => setActiveTab(item.id)}
                                className={`flex flex-col items-center gap-1 px-2 py-2 rounded-lg ${
                                    activeTab === item.id ? 'text-blue-400 bg-blue-500/20' : 'text-gray-400'
                                }`}>
                                <item.icon />
                                <span className="text-[10px] font-medium">{item.label}</span>
                            </button>
                        ))}
                    </div>
                </div>
            </div>
        );
    };

    ReactDOM.render(<TradingPlatform />, document.getElementById('root'));
</script>
